-- Table: public.cliente

-- DROP TABLE IF EXISTS public.cliente;

CREATE TABLE IF NOT EXISTS public.cliente
(
    id_cliente integer NOT NULL,
    nome character varying(100) COLLATE pg_catalog."default" NOT NULL,
    telefone character varying(15) COLLATE pg_catalog."default" NOT NULL,
    email character varying(255) COLLATE pg_catalog."default" NOT NULL,
    status character varying(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'A'::character varying,
    CONSTRAINT cliente_pkey PRIMARY KEY (id_cliente)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.cliente
    OWNER to postgres;

-- Table: public.produto

-- DROP TABLE IF EXISTS public.produto;

CREATE TABLE IF NOT EXISTS public.produto
(
    id_produto integer NOT NULL,
    descricao character varying(100) COLLATE pg_catalog."default",
    valor character varying(15) COLLATE pg_catalog."default" NOT NULL,
    categoria character varying(100) COLLATE pg_catalog."default" NOT NULL,
    status character varying(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'A'::character varying,
    CONSTRAINT produto_pkey PRIMARY KEY (id_produto)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.produto
    OWNER to postgres;
-- DROP TABLE IF EXISTS public.venda;

CREATE TABLE IF NOT EXISTS public.venda
(
    id_venda bigint NOT NULL GENERATED ALWAYS AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),
    observacoes character varying(200) COLLATE pg_catalog."default",
    data timestamp without time zone NOT NULL,
    total numeric(12,2) NOT NULL,
    status character varying(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'A'::character varying,
    cliente_id integer NOT NULL,
    CONSTRAINT venda_pkey PRIMARY KEY (id_venda),
    CONSTRAINT fk_cliente_id FOREIGN KEY (cliente_id)
        REFERENCES public.cliente (id_cliente) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.venda
    OWNER to postgres;
-- Table: public.item_venda

-- DROP TABLE IF EXISTS public.item_venda;

CREATE TABLE IF NOT EXISTS public.item_venda
(
    id_item_venda bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),
    quantidade integer NOT NULL,
    valor_unitario numeric(12,2) NOT NULL,
    valor_total numeric(12,2) NOT NULL,
    venda_id integer NOT NULL,
    produto_id integer NOT NULL,
    CONSTRAINT item_venda_pkey PRIMARY KEY (id_item_venda),
    CONSTRAINT fk_produto_id FOREIGN KEY (produto_id)
        REFERENCES public.produto (id_produto) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT fk_venda_id FOREIGN KEY (venda_id)
        REFERENCES public.venda (id_venda) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.item_venda
    OWNER to postgres;
